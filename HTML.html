import requests from bs4 import BeautifulSoup url = 'https://blackownedassociation.com/' response = requests.get(url) soup = BeautifulSoup(response.content, 'html.parser') # Find all business links on the page business_links = soup.find_all('a', {'class': 'read-more'}) # Loop through each business link and extract information for link in business_links: business_url = link['href'] business_name = link.text.strip() print('Business Name:', business_name) print('Business URL:', business_url) print('-----------------------')
